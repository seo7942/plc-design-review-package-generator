(* MASTER SAFETY CHAIN *)
(* Review-only ST Draft. 실제 접점 방향과 안전회로는 현장 검수 필요. *)

M_AGI40_MASTER_SAFETY_OK := (ESTOP_OK AND T1_PUMP_1_OVERLOAD_TRIP AND T1_PUMP_2_OVERLOAD_TRIP AND T1_PUMP_3_OVERLOAD_TRIP AND T2_PUMP_1_OVERLOAD_TRIP AND T2_PUMP_2_OVERLOAD_TRIP AND HMI_RESUME_CONFIRM_PB AND POWER_RECOVERY_DETECTED AND MCC_MAIN_POWER_OK AND UPS_POWER_OK AND T1_LEVEL_LOW_SW AND T1_LEVEL_HIGH_HIGH_SW AND T2_LEVEL_LOW_SW AND T2_LEVEL_HIGH_HIGH_SW AND T1_PUMP_1_FAULT_FB AND T1_PUMP_1_MCC_CB_OK AND T1_PUMP_1_MAINT_KEY_ON AND T1_PUMP_2_FAULT_FB AND T1_PUMP_2_MCC_CB_OK AND T1_PUMP_2_MAINT_KEY_ON AND T1_PUMP_3_FAULT_FB AND T1_PUMP_3_MCC_CB_OK AND T1_PUMP_3_MAINT_KEY_ON AND T2_PUMP_1_FAULT_FB AND T2_PUMP_1_MCC_CB_OK AND T2_PUMP_1_MAINT_KEY_ON AND T2_PUMP_2_FAULT_FB AND T2_PUMP_2_MCC_CB_OK AND T2_PUMP_2_MAINT_KEY_ON AND T1_DRAIN_VALVE_FAULT_FB AND T1_ISOLATION_VALVE_FAULT_FB AND T1_BYPASS_VALVE_FAULT_FB AND T2_TRANSFER_VALVE_FAULT_FB AND T2_ISOLATION_VALVE_FAULT_FB AND WWTP_DISCHARGE_VALVE_FAULT_FB AND LEAK_T1_AREA_DETECTED AND LEAK_T2_AREA_DETECTED AND LEAK_T1_TRENCH_DETECTED AND LEAK_T2_TRENCH_DETECTED AND LEAK_MCC_ROOM_DETECTED AND LEAK_VALVE_PIT_DETECTED AND T1_PUMP_1_FLOW_OK AND T1_PUMP_2_FLOW_OK AND T1_PUMP_3_FLOW_OK AND T2_PUMP_1_FLOW_OK AND T2_PUMP_2_FLOW_OK AND T1_LEVEL_TRANSMITTER_FAULT AND T2_LEVEL_TRANSMITTER_FAULT) AND NOT M_AGI40_SAFETY_FAULT_LATCH;

IF NOT ESTOP_OK_ANY THEN
    M_AGI40_SAFETY_FAULT_LATCH := TRUE;
END_IF;

IF NOT DOOR_CLOSED_ANY THEN
    M_AGI40_SAFETY_FAULT_LATCH := TRUE;
END_IF;

IF NOT SAFETY_RELAY_OK THEN
    M_AGI40_SAFETY_FAULT_LATCH := TRUE;
END_IF;

IF OVERLOAD_ANY THEN
    M_AGI40_SAFETY_FAULT_LATCH := TRUE;
END_IF;


(* FAULT LATCH AND MANUAL RESET *)

IF ALARM_RESET_PB AND ALARM_ACK_PB AND T1_PUMP_1_FAULT_FB AND T1_PUMP_1_OVERLOAD_TRIP AND T1_PUMP_2_FAULT_FB AND T1_PUMP_2_OVERLOAD_TRIP AND T1_PUMP_3_FAULT_FB AND T1_PUMP_3_OVERLOAD_TRIP THEN
    M_AGI40_FAULT_LATCH := TRUE;
END_IF;

M_AGI40_RESET_OK := ALARM_RESET_PB AND NOT FAULT_CONDITION_ANY;

IF M_AGI40_RESET_OK THEN
    M_AGI40_FAULT_LATCH := FALSE;
END_IF;


(* SAFETY / PROCESS FAULT SPLIT *)

SAFETY_FAULT_ANY := ESTOP_OK OR T1_PUMP_1_OVERLOAD_TRIP OR T1_PUMP_2_OVERLOAD_TRIP OR T1_PUMP_3_OVERLOAD_TRIP OR T2_PUMP_1_OVERLOAD_TRIP OR T2_PUMP_2_OVERLOAD_TRIP OR HMI_RESUME_CONFIRM_PB OR POWER_RECOVERY_DETECTED OR MCC_MAIN_POWER_OK OR UPS_POWER_OK OR T1_LEVEL_LOW_SW OR T1_LEVEL_HIGH_HIGH_SW OR T2_LEVEL_LOW_SW OR T2_LEVEL_HIGH_HIGH_SW OR T1_PUMP_1_FAULT_FB OR T1_PUMP_1_MCC_CB_OK OR T1_PUMP_1_MAINT_KEY_ON OR T1_PUMP_2_FAULT_FB OR T1_PUMP_2_MCC_CB_OK OR T1_PUMP_2_MAINT_KEY_ON OR T1_PUMP_3_FAULT_FB OR T1_PUMP_3_MCC_CB_OK OR T1_PUMP_3_MAINT_KEY_ON OR T2_PUMP_1_FAULT_FB OR T2_PUMP_1_MCC_CB_OK OR T2_PUMP_1_MAINT_KEY_ON OR T2_PUMP_2_FAULT_FB OR T2_PUMP_2_MCC_CB_OK OR T2_PUMP_2_MAINT_KEY_ON OR T1_DRAIN_VALVE_FAULT_FB OR T1_ISOLATION_VALVE_FAULT_FB OR T1_BYPASS_VALVE_FAULT_FB OR T2_TRANSFER_VALVE_FAULT_FB OR T2_ISOLATION_VALVE_FAULT_FB OR WWTP_DISCHARGE_VALVE_FAULT_FB OR LEAK_T1_AREA_DETECTED OR LEAK_T2_AREA_DETECTED OR LEAK_T1_TRENCH_DETECTED OR LEAK_T2_TRENCH_DETECTED OR LEAK_MCC_ROOM_DETECTED OR LEAK_VALVE_PIT_DETECTED OR T1_PUMP_1_FLOW_OK OR T1_PUMP_2_FLOW_OK OR T1_PUMP_3_FLOW_OK OR T2_PUMP_1_FLOW_OK OR T2_PUMP_2_FLOW_OK OR T1_LEVEL_TRANSMITTER_FAULT OR T2_LEVEL_TRANSMITTER_FAULT;
PROCESS_FAULT_ANY := ALARM_RESET_PB OR ALARM_ACK_PB OR T1_LEVEL_MID_SW OR T1_LEVEL_HIGH_SW OR T2_LEVEL_MID_SW OR T2_LEVEL_HIGH_SW OR T1_PUMP_1_RUN_FB OR T1_PUMP_2_RUN_FB OR T1_PUMP_3_RUN_FB OR T2_PUMP_1_RUN_FB OR T2_PUMP_2_RUN_FB OR T1_DRAIN_VALVE_OPEN_FB;

IF SAFETY_FAULT_ANY THEN
    M_AGI40_SAFETY_FAULT_LATCH := TRUE;
END_IF;

IF PROCESS_FAULT_ANY THEN
    M_AGI40_PROCESS_FAULT_LATCH := TRUE;
END_IF;

M_AGI40_SAFETY_RESET_OK := ALARM_RESET_PB AND NOT SAFETY_FAULT_ANY;
M_AGI40_PROCESS_RESET_OK := ALARM_RESET_PB AND NOT PROCESS_FAULT_ANY AND NOT M_AGI40_SAFETY_FAULT_LATCH;

IF M_AGI40_SAFETY_RESET_OK THEN
    M_AGI40_SAFETY_FAULT_LATCH := FALSE;
END_IF;

IF M_AGI40_PROCESS_RESET_OK THEN
    M_AGI40_PROCESS_FAULT_LATCH := FALSE;
END_IF;

(* Reset Blocked Reason 후보 *)
M_AGI40_RESET_BLOCKED_REASON_SAFETY_ACTIVE := ALARM_RESET_PB AND M_AGI40_SAFETY_FAULT_LATCH;
M_AGI40_RESET_BLOCKED_REASON_PROCESS_ACTIVE := ALARM_RESET_PB AND PROCESS_FAULT_ANY;


(* AUTO / MANUAL MODE GATE *)

MODE_CONFLICT_ALARM := PANEL_AUTO_SEL AND PANEL_MANUAL_SEL;

M_AGI40_MODE_VALID := (PANEL_AUTO_SEL AND NOT PANEL_MANUAL_SEL)
           OR (PANEL_MANUAL_SEL AND NOT PANEL_AUTO_SEL);

M_AGI40_MANUAL_PERMISSIVE := PANEL_MANUAL_SEL AND M_AGI40_MASTER_SAFETY_OK;


(* MAINTENANCE / OVERRIDE MODE GATE *)

M_AGI40_MAINTENANCE_ACTIVE := T1_PUMP_1_MAINT_KEY_ON;
M_AGI40_OVERRIDE_ACTIVE := T1_BYPASS_VALVE_OPEN_FB;

M_AGI40_OVERRIDE_PERMISSIVE := M_AGI40_OVERRIDE_ACTIVE
                      AND M_AGI40_MASTER_SAFETY_OK
                      AND NOT M_AGI40_SAFETY_FAULT_LATCH;

M_AGI40_OUTPUT_FORCE_BLOCK := (M_AGI40_MAINTENANCE_ACTIVE AND NOT M_AGI40_OVERRIDE_PERMISSIVE)
                     OR M_AGI40_SAFETY_FAULT_LATCH;

IF M_AGI40_OUTPUT_FORCE_BLOCK THEN
    MOTOR_OUTPUT_PERMISSIVE := FALSE;
END_IF;


(* SEQUENCE STEP MANAGER *)

IF M_AGI40_STEP_IDLE AND M_AGI40_MASTER_SAFETY_OK AND READY_OK AND NOT M_AGI40_SAFETY_FAULT_LATCH THEN
    M_AGI40_STEP_READY := TRUE;
    M_AGI40_STEP_IDLE := FALSE;
END_IF;

IF M_AGI40_STEP_READY AND START_PB AND NOT M_AGI40_PROCESS_FAULT_LATCH THEN
    M_AGI40_STEP_RUN := TRUE;
    M_AGI40_STEP_READY := FALSE;
END_IF;

IF M_AGI40_STEP_RUN AND M_AGI40_STEP_DONE THEN
    M_AGI40_STEP_COMPLETE := TRUE;
    M_AGI40_STEP_RUN := FALSE;
END_IF;

IF M_AGI40_STEP_RUN AND NOT M_AGI40_STEP_DONE THEN
    (* TON T_STEP_TIMEOUT, PT := T#10s *)
    IF T_STEP_TIMEOUT_Q THEN
        M_AGI40_STEP_TIMEOUT_FAULT := TRUE;
    END_IF;
END_IF;

IF M_AGI40_SAFETY_FAULT_LATCH OR M_AGI40_PROCESS_FAULT_LATCH OR M_AGI40_STEP_TIMEOUT_FAULT THEN
    M_AGI40_STEP_FAULT := TRUE;
    M_AGI40_STEP_RUN := FALSE;
END_IF;

IF STOP_PB THEN
    M_AGI40_STEP_RUN := FALSE;
END_IF;

IF ALARM_RESET_PB AND NOT M_AGI40_SAFETY_FAULT_LATCH AND NOT M_AGI40_PROCESS_FAULT_LATCH THEN
    M_AGI40_STEP_IDLE := TRUE;
    M_AGI40_STEP_FAULT := FALSE;
    M_AGI40_STEP_COMPLETE := FALSE;
END_IF;


(* MOTOR OUTPUT PERMISSIVE *)

M_AGI40_MODE_VALID := (PANEL_AUTO_SEL AND NOT PANEL_MANUAL_SEL)
           OR (PANEL_MANUAL_SEL AND NOT PANEL_AUTO_SEL);

MOTOR_OUTPUT_PERMISSIVE := M_AGI40_MASTER_SAFETY_OK
                        AND READY_OK
                        AND NOT M_AGI40_FAULT_LATCH
                        AND NOT M_AGI40_PROCESS_FAULT_LATCH
                        AND M_AGI40_MODE_VALID;

(* 대상 출력 후보: T1_PUMP_1_RUN_CMD, T1_PUMP_2_RUN_CMD, T1_PUMP_3_RUN_CMD, T2_PUMP_1_RUN_CMD, T2_PUMP_2_RUN_CMD *)


(* MOTOR FEEDBACK MONITORING *)

IF T1_PUMP_1_RUN_CMD AND NOT T1_PUMP_1_RUN_FB THEN
    (* TON T_MOTOR_START_TIMEOUT, PT := T#3s *)
    IF T_MOTOR_START_TIMEOUT_Q THEN
        M_AGI40_MOTOR_START_TIMEOUT_FAULT := TRUE;
    END_IF;
END_IF;

IF NOT T1_PUMP_1_RUN_CMD AND T1_PUMP_1_RUN_FB THEN
    (* TON T_MOTOR_STOP_TIMEOUT, PT := T#3s *)
    IF T_MOTOR_STOP_TIMEOUT_Q THEN
        M_AGI40_MOTOR_STOP_TIMEOUT_FAULT := TRUE;
    END_IF;
END_IF;

M_AGI40_MOTOR_FEEDBACK_FAULT := M_AGI40_MOTOR_START_TIMEOUT_FAULT OR M_AGI40_MOTOR_STOP_TIMEOUT_FAULT;

IF M_AGI40_MOTOR_FEEDBACK_FAULT THEN
    M_AGI40_MOTOR_FAULT := TRUE;
    M_AGI40_PROCESS_FAULT_LATCH := TRUE;
END_IF;

IF STOP_PB THEN
    MOTOR_OUTPUT_PERMISSIVE := FALSE;
END_IF;


(* VALVE OPEN/CLOSE CONTROL *)

IF T1_DRAIN_VALVE_OPEN_CMD AND T1_DRAIN_VALVE_CLOSE_CMD THEN
    M_AGI40_VALVE_FAULT := TRUE;
END_IF;

IF T1_DRAIN_VALVE_OPEN_CMD AND NOT T1_DRAIN_VALVE_OPEN_FB THEN
    (* TON T_VALVE_OPEN_TIMEOUT, PT := T#5s *)
    IF T_VALVE_OPEN_TIMEOUT_Q THEN
        M_AGI40_VALVE_OPEN_TIMEOUT_FAULT := TRUE;
    END_IF;
END_IF;

IF T1_DRAIN_VALVE_CLOSE_CMD AND NOT T1_DRAIN_VALVE_CLOSE_FB THEN
    (* TON T_VALVE_CLOSE_TIMEOUT, PT := T#5s *)
    IF T_VALVE_CLOSE_TIMEOUT_Q THEN
        M_AGI40_VALVE_CLOSE_TIMEOUT_FAULT := TRUE;
    END_IF;
END_IF;

IF T1_DRAIN_VALVE_OPEN_FB AND T1_DRAIN_VALVE_CLOSE_FB THEN
    M_AGI40_VALVE_POSITION_CONFLICT_FAULT := TRUE;
END_IF;

M_AGI40_VALVE_FAULT := M_AGI40_VALVE_FAULT
              OR M_AGI40_VALVE_OPEN_TIMEOUT_FAULT
              OR M_AGI40_VALVE_CLOSE_TIMEOUT_FAULT
              OR M_AGI40_VALVE_POSITION_CONFLICT_FAULT;

IF M_AGI40_VALVE_FAULT THEN
    M_AGI40_PROCESS_FAULT_LATCH := TRUE;
END_IF;


(* ANALOG RANGE / DEVIATION / RAMP CHECK *)

IF T1_LEVEL_AI < D_ANALOG_LOW_LIMIT THEN
    M_AGI40_ANALOG_RANGE_ALARM := TRUE;
END_IF;

IF T1_LEVEL_AI > D_ANALOG_HIGH_LIMIT THEN
    M_AGI40_ANALOG_RANGE_ALARM := TRUE;
END_IF;

IF ABS(T1_LEVEL_AI - D_ANALOG_TARGET_VALUE) > D_ANALOG_DEVIATION_LIMIT THEN
    M_AGI40_ANALOG_DEVIATION_ALARM := TRUE;
END_IF;

IF ABS(T1_LEVEL_AI - D_ANALOG_PREVIOUS_VALUE) > D_ANALOG_RAMP_RATE_LIMIT THEN
    M_AGI40_ANALOG_RAMP_RATE_ALARM := TRUE;
END_IF;

IF (T1_LEVEL_AI <= D_ANALOG_RAW_MIN) OR (T1_LEVEL_AI >= D_ANALOG_RAW_MAX) THEN
    M_AGI40_ANALOG_SENSOR_FAULT := TRUE;
END_IF;

M_AGI40_ANALOG_FAULT := M_AGI40_ANALOG_RANGE_ALARM
               OR M_AGI40_ANALOG_DEVIATION_ALARM
               OR M_AGI40_ANALOG_RAMP_RATE_ALARM
               OR M_AGI40_ANALOG_SENSOR_FAULT;

IF M_AGI40_ANALOG_FAULT THEN
    M_AGI40_PROCESS_FAULT_LATCH := TRUE;
END_IF;


(* TIMER TIMEOUT FAULT *)

IF M_AGI40_STEP_ACTIVE AND NOT STEP_DONE THEN
    (* TON T_STEP_TIMEOUT, PT := T#10s *)
    IF T_STEP_TIMEOUT_Q THEN
        M_AGI40_STEP_TIMEOUT_FAULT := TRUE;
    END_IF;
END_IF;
